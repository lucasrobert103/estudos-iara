  • Exercício 5

    No node/js, usando o terminal, criar um programa que, dado um número, exibe no terminal se é par ou ímpar.

              
            // biblioteca readline do Node.js para interagir com a entrada e saída de dados do terminal
            const readline = require('readline');
            
            // Cria uma interface readline para ler dados
            const rl = readline.createInterface({
              input: process.stdin, // Define a entrada de dados como o processo do terminal
              output: process.stdout // Define a saída de dados como o processo do terminal
            });
            
            // Pergunta ao usuário para digitar um número no terminal e chama uma função de retorno de chamada quando o usuário pressionar Enter
            rl.question('Digite um número: ', (numero) => {
              // Verifica se o número é par ou ímpar
              if(numero % 2 === 0) {
                console.log(`${numero} é par`); // Exibe uma mensagem informando que o número é par
              } else {
                console.log(`${numero} é ímpar`); // Exibe uma mensagem informando que o número é ímpar
              }
              
              // Fecha a interface readline
              rl.close();
            });

  • Exercício 6

    No node/js, usando o terminal, criar um programa que, dado uma data, exibe no terminal qual estação do ano no Brasil da data.

              
            // as estações do ano são geralmente definidas da seguinte forma:

            // Verão: de 21 de dezembro a 20 de março
            // Outono: de 21 de março a 20 de junho
            // Inverno: de 21 de junho a 20 de setembro
            // Primavera: de 21 de setembro a 20 de dezembro
            
            const readline = require('readline');
            
            function getSeason(date) {
              const month = date.getMonth() + 1;
              if (month >= 1 && month <= 3) {
                return 'Verão';
              } else if (month >= 4 && month <= 6) {
                return 'Outono';
              } else if (month >= 7 && month <= 9) {
                return 'Inverno';
              } else {
                return 'Primavera';
              }
            }
            
            const rl = readline.createInterface({
              input: process.stdin,
              output: process.stdout
            });
            
            rl.question('Digite uma data no formato DD/MM/AAAA: ', (input) => {
              const [day, month, year] = input.split('/');
              const date = new Date(year, month - 1, day);
              const season = getSeason(date);
              console.log(`A estação do ano no Brasil em ${input} é: ${season}`);
              rl.close();
            });

  • Exercício 8

    No node/js, usando o terminal, criar um programa que, dado uma data para uma função, retorna quantos dias se passaram desde o ano novo.

              
            // Pede ao usuário para digitar uma data no formato "DD/MM/AAAA"
            const readline = require('readline').createInterface({
              input: process.stdin,
              output: process.stdout
            });
            
            readline.question('Digite uma data no formato "DD/MM/AAAA": ', (date) => {
              // Converte a data para um objeto Date
              const parts = date.split('/');
              const d = new Date(`${parts[2]}-${parts[1]}-${parts[0]}`);
            
              // Calcula o número de dias desde o início do ano
              const startOfYear = new Date(d.getFullYear(), 0, 1);
              const diff = (d - startOfYear) / (1000 * 60 * 60 * 24);
            
              console.log(`Dias desde o início do ano: ${diff}`);
            
              readline.close();
            });

  • Exercício 9

    No node/js, usando o terminal, criar um programa que, dado uma data de nascimento para uma função, retorna quantos anos tem o indivíduo.

              
            // Pede ao usuário para digitar uma data de nascimento no formato "DD/MM/AAAA"
            const readline = require('readline').createInterface({
              input: process.stdin,
              output: process.stdout
            });
            
            readline.question('Digite uma data de nascimento no formato "DD/MM/AAAA": ', (birthdate) => {
              // Converte a data de nascimento para um objeto Date
              const parts = birthdate.split('/');
              const dob = new Date(`${parts[2]}-${parts[1]}-${parts[0]}`);
            
              // Calcula a idade em anos
              const now = new Date();
              let age = now.getFullYear() - dob.getFullYear();
              if (now.getMonth() < dob.getMonth() || (now.getMonth() === dob.getMonth() && now.getDate() < dob.getDate())) {
                age--;
              }
            
              console.log(`Idade: ${age} anos`);
            
              readline.close();
            });

  • Exercício 14

    Fazer um HTML com um botão e um input. Dado um número qualquer no input, ao clicar no botão exibir na tela: O seu valor absoluto, valor inteiro mais próximo, raiz quadrada é do número, raiz cúbica do número, número ao quadrado

              
            <!DOCTYPE html>
            <html>
            <head>
                <title>Exemplo de botão e input com resultados</title>
            </head>
            <body>
                <label for="numero">Digite um número:</label>
                <input type="number" id="numero">
            
                <button onclick="exibirResultados()">Clique aqui</button>
            
                <div id="resultados"></div>
            
                <script>
                    function exibirResultados() {
                        const numero = Number(document.getElementById("numero").value);
                        const valorAbsoluto = Math.abs(numero);
                        const inteiroProximo = Math.round(numero);
                        const raizQuadrada = Math.sqrt(numero);
                        const raizCubica = Math.cbrt(numero);
                        const quadrado = numero ** 2;
            
                        const resultadosDiv = document.getElementById("resultados");
                        resultadosDiv.innerHTML = `
                            <p>Valor absoluto: ${valorAbsoluto}</p>
                            <p>Inteiro mais próximo: ${inteiroProximo}</p>
                            <p>Raiz quadrada: ${raizQuadrada}</p>
                            <p>Raiz cúbica: ${raizCubica}</p>
                            <p>Ao quadrado: ${quadrado}</p>
                        `;
                    }
                </script>
            </body>
            </html>

  • Exercício 16

    Fazer HTML com 2 botões e 1 input. O primeiro botão adiciona itens numa lista de números, sendo o valor vindo do input. O segundo botão calcula...

              
            <!DOCTYPE html>
            <html>
            <head>
                <title>Lista de números e média</title>
            </head>
            <body>
                <h1>Lista de números</h1>
            
                <label for="numero">Número:</label>
                <input type="number" id="numero">
                <button onclick="adicionarNumero()">Adicionar número</button>
            
                <ul id="listaNumeros"></ul>
            
                <button onclick="calcularMedia()">Calcular média</button>
            
                <p id="media"></p>
            
                <script>
                    let numeros = [];
            
                    function adicionarNumero() {
                        let numero = document.getElementById("numero").value;
                        numeros.push(numero);
                        document.getElementById("listaNumeros").innerHTML = "";
                        numeros.forEach(function(numero) {
                            document.getElementById("listaNumeros").innerHTML += "<li>" + numero + "</li>";
                        });
                        document.getElementById("numero").value = "";
                    }
            
                    function calcularMedia() {
                        let soma = 0;
                        numeros.forEach(function(numero) {
                            soma += parseInt(numero);
                        });
                        let media = soma / numeros.length;
                        document.getElementById("media").textContent = "Média: " + media.toFixed(2);
                    }
                </script>
            </body>
            </html>
